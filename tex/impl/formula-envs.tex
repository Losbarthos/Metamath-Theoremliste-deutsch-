\DeclareRobustCommand{\ShowFormulaSource}[1]{%
  \ifmmode
    \text{\normalfont\ttfamily\footnotesize\detokenize{#1}}%
  \else
    {\normalfont\ttfamily\footnotesize\detokenize{#1}}%
  \fi
}

\ExplSyntaxOn
\seq_new:N \g_mx_formula_stack_seq

% Stack-Eintrag: {env}{label}{formula}{id}
\cs_set_protected:Npn \mx_push_formula:nnnn #1#2#3#4
  {
    \tl_set:Nx \l_tmpa_tl {#2}
    \tl_set:Nn \l_tmpb_tl {#3}
    \tl_set:Nn \l_tmpc_tl {#4}
    \seq_gput_right:Nx \g_mx_formula_stack_seq
      {
        {\exp_not:n{#1}}
        {\exp_not:V \l_tmpa_tl}
        {\exp_not:V \l_tmpb_tl}
        {\exp_not:V \l_tmpc_tl}
      }
  }

\NewDocumentCommand{\MxPushFormula}{m m m o}
  {
    \IfNoValueTF{#4}
      { \mx_push_formula:nnnn {#1}{#2}{#3} {} }
      { \mx_push_formula:nnnn {#1}{#2}{#3} {#4} }
  }

\NewDocumentCommand{\MxFormulaStackIdRef}{m m m}{%
  \IfBlankTF{#3}
    {\ThmLookupEmitRef{#1}{#2}}
    {\hyperref[#2]{\ShowFormulaSource{#3}}}%
}

% ------------------------------------------------------------
% Einzige öffentliche Makros
% Signatur: \FormulaThm[<Titel>]{<Formel>}{<ID>}{<Delta-Zeilen>}
%           \FormulaDef[<Titel>]{<Formel>}{<ID>}{<Delta-Zeilen>}
%           \FormulaAxiom[<Titel>]{<Formel>}{<ID>}{<Delta-Zeilen>}
% Der letzte Block darf leer sein: {}
% ------------------------------------------------------------
\NewDocumentCommand{\FormulaThm}{o +m m +m}{%
  \IfNoValueTF{#1}
    {\begin{formulaThm}}
    {\begin{formulaThm}[#1]}%
  \edef\mxLbl{mx:theorem:#3}%
  \phantomsection\label{\mxLbl}%
  \IfNoValueTF{#1}
    {\ThmLookupRegister{\mxLbl}{theorem}{\theformulaThm}{#2}}
    {\ThmLookupRegister{\mxLbl}{theorem}{\theformulaThm}{#2}{#1}}%
  \ThmLookupRegisterId{#3}{theorem}{\mxLbl}%
  \MxPushFormula{theorem}{\mxLbl}{#2}[#3]%
  \IfBlankTF{#4}{}{\DeltaBlockFromRows{#4}}%
  \MxDisplayFormulaWithId{#2}{#3}%
  \end{formulaThm}%
}

\NewDocumentCommand{\FormulaDef}{o +m m +m}{%
  \IfNoValueTF{#1}
    {\begin{formulaDef}}
    {\begin{formulaDef}[#1]}%
  \edef\mxLbl{mx:definition:#3}%
  \phantomsection\label{\mxLbl}%
  \IfNoValueTF{#1}
    {\ThmLookupRegister{\mxLbl}{definition}{\theformulaDef}{#2}}
    {\ThmLookupRegister{\mxLbl}{definition}{\theformulaDef}{#2}{#1}}%
  \ThmLookupRegisterId{#3}{definition}{\mxLbl}%
  \MxPushFormula{definition}{\mxLbl}{#2}[#3]%
  \IfBlankTF{#4}{}{\DeltaBlockFromRows{#4}}%
  \MxDisplayFormulaWithId{#2}{#3}%
  \end{formulaDef}%
}

\NewDocumentCommand{\FormulaAxiom}{o +m m +m}{%
  \IfNoValueTF{#1}
    {\begin{formulaAx}}
    {\begin{formulaAx}[#1]}%
  \edef\mxLbl{mx:axiom:#3}%
  \phantomsection\label{\mxLbl}%
  \IfNoValueTF{#1}
    {\ThmLookupRegister{\mxLbl}{axiom}{\theformulaAx}{#2}}
    {\ThmLookupRegister{\mxLbl}{axiom}{\theformulaAx}{#2}{#1}}%
  \ThmLookupRegisterId{#3}{axiom}{\mxLbl}%
  \MxPushFormula{axiom}{\mxLbl}{#2}[#3]%
  \IfBlankTF{#4}{}{\DeltaBlockFromRows{#4}}%

  \end{formulaAx}%
}

\ExplSyntaxOn
\NewDocumentCommand{\FormulaRefAuto}{s O{} m o g}{%
  \tl_set:Nn \l_tmpa_tl {#2}

  \IfNoValueF{#4}{
    \str_case:nn {#4}{
      {ax}  { \tl_put_right:Nn \l_tmpa_tl { env=axiom } }
      {def} { \tl_put_right:Nn \l_tmpa_tl { env=definition } }
      {thm} { \tl_put_right:Nn \l_tmpa_tl { env=theorem } }
    }{}
  }

  \regex_match:nnTF { \A [A-Za-z0-9:._\-]+ \Z } {#3}
    {
      \ThmRefById{#3}
    }
    {
      \IfBooleanTF{#1}
        { \ThmRefByStructure*[\tl_use:N \l_tmpa_tl]{#3} }
        { \ThmRefByStructure [\tl_use:N \l_tmpa_tl]{#3} }
    }

  \mx_formula_ref_suffix:n {#5}
}
